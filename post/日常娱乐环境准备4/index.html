6

8

#加入下面内如

}
#重新加载
[root@localhost ~]# systemctl daemon-reload 
[root@localhost ~]# systemctl reload docker.service

如果启动kube-apiserver.service 时报错

1
2

#方法1
[root@localhost ~]# systemctl start kube-apiserver.service 
Job for kube-apiserver.service failed because the control process exited with error code. See "systemctl status kube-apiserver.service" and "journalctl -xe" for details.
#可能是因为8080端口被占用导致的,需要修改/etc/kubernetes/apiserver 的
KUBE_API_PORT="--port=8080"  将前面的#号删除,把端口修改为空闲端口号
#重启服务
[root@localhost kubernetes]# systemctl restart kube-apiserver.service
